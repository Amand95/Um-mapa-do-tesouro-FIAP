-- Tabela para Culturas
CREATE TABLE Cultura (
    Cultura_ID INT PRIMARY KEY,
    Nome_Cultura VARCHAR(255) NOT NULL
);

-- Tabela para Sensores
CREATE TABLE Sensor (
    Sensor_ID INT PRIMARY KEY,
    Tipo_Sensor VARCHAR(50) NOT NULL, 
    Unidade_Medida VARCHAR(50) NOT NULL
);

-- Tabela para Medições
CREATE TABLE Medicao (
    Medicao_ID INT PRIMARY KEY,
    Sensor_ID INT,
    Cultura_ID INT,
    Data_Hora TIMESTAMP NOT NULL,
    Valor DECIMAL(10,2),
    FOREIGN KEY (Sensor_ID) REFERENCES Sensor(Sensor_ID),
    FOREIGN KEY (Cultura_ID) REFERENCES Cultura(Cultura_ID)
);

-- Tabela para Ajustes
CREATE TABLE Ajuste (
    Ajuste_ID INT PRIMARY KEY,
    Medicao_ID INT,
    Quantidade_Agua DECIMAL(10,2),
    Quantidade_Nutriente DECIMAL(10,2),
    Data_Hora TIMESTAMP NOT NULL,
    FOREIGN KEY (Medicao_ID) REFERENCES Medicao(Medicao_ID)
);
